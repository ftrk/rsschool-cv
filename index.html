    export default {
        data() {
            return {
                userInput: "",
                toDoList: [],
                doneList: [],
            };
        },
        methods: {
            addToList() {
                this.toDoList.push({
                    id: Date.now(),
                    task: this.userInput,
                });
                this.userInput = "";
            },
            removeToList(el, list) {
                list === this.toDoList
                    ? (this.toDoList = this.toDoList.filter((item) => item.id !== el.id))
                    : (this.doneList = this.doneList.filter((item) => item.id !== el.id));
            },
            changeCheckbox(el, list) {
                list === this.toDoList ? this.doneList.push(el) : this.toDoList.push(el);
                this.removeToList(el, list);
            },
        },
    };
